<template>
  <a-layout class="min-h-screen">

    <!-- Header -->
    <a-layout-header class="flex justify-between items-center px-6">

      <div class="text-white text-lg font-semibold">
        My App
      </div>

      <div class="flex items-center gap-4">

        <template v-if="auth.user">

          <span class="text-white">
            {{ auth.user.username }}
          </span>

          <a-button type="primary" ghost @click="goProfile">
            Profile
          </a-button>

          <a-button danger @click="logout">
            Logout
          </a-button>

        </template>

        <template v-else>

          <a-button type="primary" ghost @click="goLogin">
            Login
          </a-button>

          <a-button @click="goRegister">
            Register
          </a-button>

        </template>

      </div>

    </a-layout-header>

    <!-- Content -->
    <a-layout-content class="p-10">

      <router-view />

    </a-layout-content>

  </a-layout>
</template>

<script setup lang="ts">

import { onMounted } from "vue"
import { useRouter } from "vue-router"
import { useAuthStore } from "@/features/auth/store/authStore"

const router = useRouter()
const auth = useAuthStore()

onMounted(async () => {

  if (auth.token && !auth.user) {
    try {
      await auth.fetchUser()
    } catch {
      auth.logout()
    }
  }

})

function goLogin() {
  router.push("/login")
}

function goRegister() {
  router.push("/register")
}

function goProfile() {
  router.push("/profile")
}

async function logout() {

  await auth.logout()

  router.push("/login")
}

</script>

import { createRouter, createWebHistory } from "vue-router"

import LoginPage from "@/features/auth/pages/LoginPage.vue"
import RegisterPage from "@/features/auth/pages/RegisterPage.vue"
import DashboardPage from "@/features/dashboard/pages/DashboardPage.vue"
import ProfilePage from "@/features/user/pages/ProfilePage.vue"

const routes = [

  { path: "/login", component: LoginPage },

  { path: "/register", component: RegisterPage },

  { path: "/", component: DashboardPage },

  { path: "/profile", component: ProfilePage }

]

export default createRouter({
  history: createWebHistory(),
  routes
})
